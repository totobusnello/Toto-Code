  stackVectors(candidates)

const gnnResult = await wrapper.gnnEnhancedSearch(query, {
  k: 10,
  graphContext: {
    nodes: documentEmbeddings,
    edges: documentRelations,
  },
});

const coordinator = new AttentionCoordinator(wrapper.getAttentionService());
